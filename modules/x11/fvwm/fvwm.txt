#
# General Settings
#

# All screens show the same desktop. Not relevant since I only have one monitor.
# per host.
DesktopConfiguration global

# Desktops and pages.
DesktopName 0 Desk0
Desktopname 1 Desk1
DesktopName 2 Desk2
DesktopName 3 Desk3
DesktopName 4 Desk4
DesktopSize 1x1

# EwmhBaseStruts [screen name] [left] [right] [top] [bottom]
# Reserves space along the edge(s) of the Screen that will not
# be covered when maximizing or placing windows.
EwmhBaseStruts screen $[monitor.primary] 0 0 48 0

# This sets the ClickTime and MoveThreshold used to determine
# Double Clicks, Hold and Move for the mouse.
ClickTime 250
MoveThreshold 3

# Window focus behaviour.
Style * MouseFocus
# Clicking on a window raises it.
Style * MouseFocusClickRaises
# Transient windows should grab focus.
Style * FPGrabFocusTransient

# Place new windows under the mouse pointer.
Style * PositionPlacement UnderMouse
# When a new window is created, it grabs focus.
Style * GrabFocusOn
# When a new window is created, ignore the app's requested
# position.
Style * !UsePPosition

# Move a wireframe of the window.
OpaqueMoveSize 0
# Resize a wireframe of the window.
Style * ResizeOutline

# When moving windows, snap to window and screen edges.
Style * SnapGrid, SnapAttraction 15 SameType ScreenAll

# Transient Windows (such as open file windows)
Style * DecorateTransient, StackTransientParent
Style * !FPGrabFocusTransient, FPReleaseFocusTransient

# WindowShade
Style * WindowShadeScrolls, WindowShadeSteps 10

# Ignore Numlock and other modifiers for bindings
# See http://fvwm.org/documentation/faq/#why-do-numlock-capslock-and-scrolllock-interfere-with-clicktofocus-andor-my-mouse-bindings
IgnoreModifiers L25

# ICCCM allows windows to tell the WM to not give them keyboard focus.
# Many windows use this incorrectly and expect keyboard focus anyways.
# This styles tells fvwm to ignore this property, which makes some
# applications (such as java and electron apps) behave better.
Style * Lenience

#
# Functions
#

# Start Function
#
# The start function is run right after fvwm is done reading
# the config file. This function run after each restart
# so using Test (Init) or Test (Restart) can further control
# actions that are run during the first time run (Init) or
# actions that are run after a restart.
DestroyFunc StartFunction
AddToFunc   StartFunction
+ I Test (Init) Exec exec feh --bg-fill $[infostore.walls]/osx-panther.png
+ I Module FvwmEvent EventNewDesk
+ I Module FvwmButtons MainPanel

#
# Decor
#

# Width, in pixels, of the 3D decor around menus.
MenuStyle * Hilight3DThickness 2
# Colorset of inactive menu items.
MenuStyle * MenuColorset 5,
# Colorset of active (hover) menu items.
MenuStyle * ActiveColorset 6
# Colorset of disabled menu items.
MenuStyle * GreyedColorset 7
# Colorset of menu titles.
MenuStyle * TitleColorset 8
# Highlight active menus items with a thick 3D border.
MenuStyle * Hilight3DThick
# Highlight the background of active menu items.
MenuStyle * HilightBack
# Highlight menu titles.
MenuStyle * HilightTitleBack
# Long separators cut across the whole width of the menu,
# short ones leave a few pixels of margin.
MenuStyle * SeparatorsShort
# Submenu triangles are filled in.
MenuStyle * TrianglesSolid
# Submenu triangles are drawn using the foreground colour of
# the MenuColorset.
MenuStyle * TrianglesUseFore
# Menu item format: picture, start of highlight area,
# mini-icon, labels, sub-menu triangle, end of highlight
# area.
MenuStyle * ItemFormat "%s%|%3.1i%5.3l%5l%5r%5.3>%|"

# Title style.

# Colorset for inactive windows.
Style * Colorset 1
# Colorset for the active window.
Style * HilightColorset 2
# Colorset for inactive window borders.
Style * BorderColorset 3
# Colorset for active window borders.
Style * HilightBorderColorset 4
# Border width.
Style * BorderWidth 8
# Resize handle width.
Style * HandleWidth 8
# Use MWM-style borders.
Style * MwmBorder
# Clicking on the border to resize does not change the
# look of the border.
Style * FirmBorder

# Button style.

ButtonStyle All Simple -- UseTitleStyle

# Shape of the "close window" button.

AddButtonStyle 2 Vector 3 21x79@1 21x21@1 79x21@1

# Border style.
BorderStyle Simple

# C-s-w: launch browser

# C-s-f: launch file manager

Silent Key Pause A N Exec exec xscreensaver-command -lock

# Print: take a screenshot of a region
Silent Key Print A N Exec exec scrot -f -s

# C-Print: take a screenshot of the current window.
Silent Key Print A C Exec exec scrot -u -b

# M-Print: take a screenshot of the whole screen.
Silent Key Print A M Exec exec scrot

# M-Tab: window list
Silent Key Tab A M WindowList Root c c NoGeometry, SelectOnRelease Alt_L, SortByClass

#
# Mouse Bindings
#

# super + left click hold: move window
Mouse 1 W 4 Move

# super + right click hold: resize window
Mouse 3 W 4 Resize

# left click on the rightmost window button: close
Mouse 1 2 A Close

# Click on the titlebar to Raise, hold to move, double click to maximize.
DestroyFunc RaiseMoveX
AddToFunc RaiseMoveX
+ I Raise
+ M $0
+ D $1

Mouse 1 T A RaiseMoveX Move Maximize

# Mouse Wheel Up/Down on titlebar to WindowShade On/Off
Mouse 4 T    A WindowShade True
Mouse 5 T    A WindowShade False

# Borders: Click to raise, Move to Resize
DestroyFunc RaiseMove
AddToFunc RaiseMove
+ I Raise
+ M $0

Mouse 1 FS A RaiseMove Resize

# Left click on the root window: nothing.
Mouse 1 R A Nop

# Right click on the root window: launch the FVWM menu.
Mouse 3 R A Menu MenuFvwmRoot

# Middle click on the root window: open the window list.
Mouse 2 R A WindowList
